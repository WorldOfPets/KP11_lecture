### –ó–∞–¥–∞—á–∞ 1: –¢–∞–±–ª–∏—Ü–∞ —É–º–Ω–æ–∂–µ–Ω–∏—è

    print()  # –ü–µ—Ä–µ—Ö–æ–¥ –Ω–∞ –Ω–æ–≤—É—é —Å—Ç—Ä–æ–∫—É –ø–æ—Å–ª–µ –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–∏ —Ç–∞–±–ª–∏—Ü—ã

### –ó–∞–¥–∞—á–∞ 2: –í—ã–≤–æ–¥ –º–∞—Ç—Ä–∏—Ü—ã

**–û–ø–∏—Å–∞–Ω–∏–µ:** –°–æ–∑–¥–∞–π—Ç–µ 3x3 –º–∞—Ç—Ä–∏—Ü—É –∏ –∑–∞–ø–æ–ª–Ω–∏—Ç–µ –µ—ë —á–∏—Å–ª–∞–º–∏ –æ—Ç 1 –¥–æ 9. –í—ã–≤–µ–¥–∏—Ç–µ –Ω–∞ —ç–∫—Ä–∞–Ω.

```python
matrix = [[0, 0, 0] for _ in range(3)]  # –°–æ–∑–¥–∞–µ–º –ø—É—Å—Ç—É—é –º–∞—Ç—Ä–∏—Ü—É 3x3
count = 1
for i in range(3):
    for j in range(3):
        matrix[i][j] = count
        count += 1

# –í—ã–≤–æ–¥ –º–∞—Ç—Ä–∏—Ü—ã
for row in matrix:
    for element in row:
        print(element, end=' ')
    print()
```

### –ó–∞–¥–∞—á–∞ 3: –ó–≤—ë–∑–¥–æ—á–∫–∏ –≤ —Ç—Ä–µ—É–≥–æ–ª—å–Ω–∏–∫–µ

### –ó–∞–¥–∞—á–∞ 4: –í–≤–æ–¥ –º–∞—Ç—Ä–∏—Ü—ã –∏ –ø–æ–¥—Å—á–µ—Ç —Å—É–º–º—ã —ç–ª–µ–º–µ–Ω—Ç–æ–≤

**–û–ø–∏—Å–∞–Ω–∏–µ:** –¢—Ä–µ–±—É–µ—Ç—Å—è —Å–æ–∑–¥–∞—Ç—å –ø—Ä–æ–≥—Ä–∞–º–º—É, –∫–æ—Ç–æ—Ä–∞—è –∑–∞–ø—Ä–∞—à–∏–≤–∞–µ—Ç —É –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è —Ä–∞–∑–º–µ—Ä—ã –º–∞—Ç—Ä–∏—Ü—ã (–∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ –∏ —Å—Ç–æ–ª–±—Ü–æ–≤), –∑–∞—Ç–µ–º –∑–∞–ø—Ä–∞—à–∏–≤–∞–µ—Ç —Å–∞–º–∏ —ç–ª–µ–º–µ–Ω—Ç—ã –º–∞—Ç—Ä–∏—Ü—ã –∏ –ø–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ—Ç –∏—Ö —Å—É–º–º—É.

```python
rows = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ –º–∞—Ç—Ä–∏—Ü—ã: "))
cols = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç–æ–ª–±—Ü–æ–≤ –º–∞—Ç—Ä–∏—Ü—ã: "))

matrix = []
for i in range(rows):
    row = []
    for j in range(cols):
        element = int(input(f"–í–≤–µ–¥–∏—Ç–µ —ç–ª–µ–º–µ–Ω—Ç [{i+1}][{j+1}]: "))
        row.append(element)
    matrix.append(row)

# –ü–æ–¥—Å—á–µ—Ç —Å—É–º–º—ã —ç–ª–µ–º–µ–Ω—Ç–æ–≤

    for element in row:
        total_sum += element

```

### –ó–∞–¥–∞—á–∞ 5: –£–Ω–∏–∫–∞–ª—å–Ω—ã–µ –ø–∞—Ä—ã —á–∏—Å–µ–ª
